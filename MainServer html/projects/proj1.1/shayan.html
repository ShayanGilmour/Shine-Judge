def getContents(file_name):
    result = []
    with open(file_name, 'r') as f:
        for line in f:
            result.append(line)

    return result

def spliter():
    global Total
    print('+', end='');
    for i in range(Total):
        print('-', end='');
    print('+', end='');
    print("");

def fix (s):
    "Remove \n"
    if (s[-1] == '\n'):
        s = s[:-1];
    return s;

def adjust(s, index):
    s = fix(s);
    return s + (" "*(Size[index]-len(s)-1))


def showLine(line):
    for i in range(len(line)):
        print(adjust(line[i], i), end='');
        print('|', end='');


MaxM = int(1e5);
Size = [int(0)]*MaxM;
Total = 1;

for i in range(MaxM):
