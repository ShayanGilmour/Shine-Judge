def getContents(file_name):
    result = []
    with open(file_name, 'r') as f:
        for line in f:
            result.append(line[:-1].split(','))

    return result
def printEmptyLine(a):
    print('+',end='')
    for i in range(a-2):
        print('-',end='')
    print('+')
inp='test1'
contents=getContents((inp+'.csv'))
width=[-1 for i in contents[0]]
fullWidth=0
for i in range(len(width)):
    for j in contents:
        if(len(j[i])>width[i]):
            width[i]=len(j[i])
    fullWidth+=width[i]
fullWidth+=len(width)+1
for content in contents:
    printEmptyLine(fullWidth)
    for i in range(len(content)):
        print("|",content[i].ljust(width[i]," "),end='',sep='')
    print('|')
printEmptyLine(fullWidth)
