def contents(file_name):
    result = []
    with open(file_name, 'r') as f:
        for line in f:
            result.append(line[:-2])

    return result
def csvExtractor(datalines):
    data = list()
    for l in datalines:
        data.append(l.split(","))
    return data
def sortCol(data):
    maxCol = [0] * len(data[0])  
    k = 0  
    for l in data:
        for i in range(len(l)):
            if maxCol[i] < len(l[i]):
                maxCol[i] = len(l[i])
        k += 1
    return maxCol
def printLine(maxcol):
    pline = "+"
    pline += "".ljust(sum(maxcol) + len(maxcol) - 1,"-") + "+"
    print(pline)
def printTitle(titles,maxcol):
    printLine(maxcol)
    pline = "|"
    for i in range(len(titles)):
        pline += titles[i] + "".ljust(maxcol[i] - len(titles[i])) + "|"

    print(pline)
    printLine(maxcol)
def printData(data,maxcol):
    data = data[1:]
    for l in data:
        pline = "|"

        for i in range(len(l)):
            pline += l[i] + "".ljust(maxcol[i] - len(l[i])) + "|"

        print(pline)
        printLine(maxcol)

filename = input()
datalines = contents(filename)
data = csvExtractor(datalines)
maxcol = sortCol(data)
printTitle(data[0],maxcol)
printData(data,maxcol)
