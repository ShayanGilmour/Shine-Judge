def find(a , list):
    for i in range(len(list)):
        if (list[i] == a):
            return i
    return -1
def countAndSplitString(string):
    global usedCharacters
    global countStr
    for i in range(len(string)):
        index = find(string[i] , usedCharacters)
        if (index != -1):
            countStr[index] +=1
        else:
            usedCharacters.append(string[i])
            countStr.append(1)
def switchElement(list , i , j):
    Tmp = list[i]
    list[i] = list[j]
    list[j] = Tmp
def sort(list1 , list2):
    for i in range(len(list1)):
        for j in range (len(list1)):
            if (list1[j] > list1[i]):
                switchElement(list1 , i , j)
                switchElement(list2 , i , j)
    return
